<script>
// ── DATA LAYER ──
const CATS = ['Bil','Hus','Disc golf','Fodbold'];
const CAT_ICONS = { 'Bil':'🚗', 'Hus':'🏠', 'Disc golf':'🥏', 'Fodbold':'⚽' };
const CAT_COLORS = { 'Bil':'#4f8ef7', 'Hus':'#2ecc8a', 'Disc golf':'#f7c94f', 'Fodbold':'#f75f5f' };
const CAT_CLASS = { 'Bil':'badge-cat1', 'Hus':'badge-cat2', 'Disc golf':'badge-cat3', 'Fodbold':'badge-cat4' };
const MONTHS = ['Januar','Februar','Marts','April','Maj','Juni','Juli','August','September','Oktober','November','December'];

function load(key, def) { try { return JSON.parse(localStorage.getItem(key)) ?? def; } catch { return def; } }
function save(key, val) { localStorage.setItem(key, JSON.stringify(val)); }

function getSetup() { return load('setup', { year: new Date().getFullYear(), budgets: { Bil:0, Hus:0, 'Disc golf':0, Fodbold:0 } }); }
function getExpenses() { return load('expenses', []); }
function saveExpenses(arr) { save('expenses', arr); }

// ── NAVIGATION ──
function showPage(name, btn) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById('page-' + name).classList.add('active');
  document.querySelectorAll('.botnav button').forEach(b => b.classList.remove('active'));
  if (btn) btn.classList.add('active');
  if (name === 'overblik') renderOverblik();
  if (name === 'udgifter') renderExpenses();
  if (name === 'maaned') renderMonthly();
  if (name === 'indstil') loadSetup();
}

// ── SETUP PAGE ──
function loadSetup() {
  const s = getSetup();
  document.getElementById('setupYear').value = s.year;
  CATS.forEach(c => { const el = document.getElementById('bud-'+c); if (el) el.value = s.budgets[c] || ''; });
}
function saveSetup() {
  const year = parseInt(document.getElementById('setupYear').value) || new Date().getFullYear();
  const budgets = {};
  CATS.forEach(c => { budgets[c] = parseFloat(document.getElementById('bud-'+c).value) || 0; });
  save('setup', { year, budgets });
  renderAll();
  showFeedback('Opsætning gemt ✓');
}

// ── OVERBLIK ──
function renderOverblik() {
  const s = getSetup();
  const expenses = getExpenses();
  document.getElementById('topYear').textContent = s.year;

  // totals
  let totalBudget = 0, totalSpent = 0;
  CATS.forEach(c => { totalBudget += s.budgets[c] || 0; });
  expenses.forEach(e => { totalSpent += e.amount; });
  const totalLeft = totalBudget - totalSpent;
  document.getElementById('sum-budget').textContent = fmt(totalBudget);
  document.getElementById('sum-spent').textContent = fmt(totalSpent);
  document.getElementById('sum-left').textContent = fmt(totalLeft);

  // bars
  const container = document.getElementById('budget-bars');
  container.innerHTML = '';
  CATS.forEach(c => {
    const budget = s.budgets[c] || 0;
    const spent = expenses.filter(e => e.cat === c).reduce((a, e) => a + e.amount, 0);
    const left = budget - spent;
    const pct = budget > 0 ? Math.min((spent / budget) * 100, 100) : 0;
    const barClass = pct < 70 ? 'bar-ok' : pct < 90 ? 'bar-warn' : 'bar-over';
    container.innerHTML += `
      <div class="budget-item">
        <div class="budget-row">
          <div class="budget-label">
            <div class="cat-dot" style="background:${CAT_COLORS[c]}"></div>
            ${CAT_ICONS[c]} ${c}
          </div>
          <div class="budget-amounts">
            <strong>${fmt(spent)}</strong> / ${fmt(budget)}
          </div>
        </div>
        <div class="bar-track"><div class="bar-fill ${barClass}" style="width:${pct}%"></div></div>
        <div class="bar-pct">${left >= 0 ? fmt(left) + ' tilbage' : fmt(Math.abs(left)) + ' over budget'} · ${pct.toFixed(0)}%</div>
      </div>`;
  });

  // pie chart
  drawPie(expenses);
}

function drawPie(expenses) {
  const canvas = document.getElementById('pieChart');
  const ctx = canvas.getContext('2d');
  const size = 220;
  canvas.width = size; canvas.height = size;
  ctx.clearRect(0, 0, size, size);

  const totals = {};
  CATS.forEach(c => totals[c] = 0);
  expenses.forEach(e => { if (totals[e.cat] !== undefined) totals[e.cat] += e.amount; });
  const total = Object.values(totals).reduce((a,b) => a+b, 0);

  const legend = document.getElementById('chart-legend');
  legend.innerHTML = '';

  if (total === 0) {
    ctx.beginPath();
    ctx.arc(size/2, size/2, size/2 - 10, 0, Math.PI*2);
    ctx.fillStyle = '#22263a';
    ctx.fill();
    ctx.fillStyle = '#8b90b0';
    ctx.font = '13px Georgia';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText('Ingen data', size/2, size/2);
    return;
  }

  let startAngle = -Math.PI / 2;
  CATS.forEach(c => {
    const val = totals[c];
    if (val === 0) return;
    const slice = (val / total) * Math.PI * 2;
    ctx.beginPath();
    ctx.moveTo(size/2, size/2);
    ctx.arc(size/2, size/2, size/2 - 6, startAngle, startAngle + slice);
    ctx.closePath();
    ctx.fillStyle = CAT_COLORS[c];
    ctx.fill();
    ctx.strokeStyle = '#0f1117';
    ctx.lineWidth = 3;
    ctx.stroke();
    startAngle += slice;
    const pct = ((val/total)*100).toFixed(0);
    legend.innerHTML += `<div class="legend-item"><div class="legend-dot" style="background:${CAT_COLORS[c]}"></div>${CAT_ICONS[c]} ${c} – ${pct}%</div>`;
  });

  // center hole
  ctx.beginPath();
  ctx.arc(size/2, size/2, size/2 - 48, 0, Math.PI*2);
  ctx.fillStyle = '#1a1d27';
  ctx.fill();
  ctx.fillStyle = '#e8eaf6';
  ctx.font = 'bold 14px Georgia';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText(fmt(total), size/2, size/2);
}

// ── EXPENSE LIST ──
function renderExpenses() {
  const expenses = getExpenses();
  const q = (document.getElementById('searchInput')?.value || '').toLowerCase();
  const cat = document.getElementById('filterCat')?.value || '';
  const filtered = expenses.filter(e => {
    if (cat && e.cat !== cat) return false;
    if (q && !(e.desc.toLowerCase().includes(q) || e.cat.toLowerCase().includes(q))) return false;
    return true;
  }).sort((a,b) => b.date.localeCompare(a.date));

  const el = document.getElementById('expense-list');
  if (filtered.length === 0) { el.innerHTML = '<div class="empty">Ingen udgifter endnu</div>'; return; }

  el.innerHTML = filtered.map(e => `
    <div class="expense-item">
      <div class="cat-dot expense-cat-dot" style="background:${CAT_COLORS[e.cat]}"></div>
      <div class="expense-info">
        <div class="expense-desc">${e.desc || e.cat}</div>
        <div class="expense-meta">${fmtDate(e.date)} · <span class="badge ${CAT_CLASS[e.cat]}">${e.cat}</span></div>
      </div>
      <div class="expense-amount">${fmt(e.amount)}</div>
      <div class="expense-actions">
        <button class="btn-small btn-edit" onclick="openEditModal('${e.id}')">✏️</button>
        <button class="btn-small btn-del" onclick="deleteExpense('${e.id}')">🗑️</button>
      </div>
    </div>`).join('');
}

// ── MONTHLY VIEW ──
function renderMonthly() {
  const expenses = getExpenses();
  const s = getSetup();
  const byMonth = {};
  expenses.forEach(e => {
    const m = parseInt(e.date.split('-')[1]) - 1;
    if (!byMonth[m]) byMonth[m] = { total: 0, cats: {} };
    byMonth[m].total += e.amount;
    byMonth[m].cats[e.cat] = (byMonth[m].cats[e.cat] || 0) + e.amount;
  });

  const el = document.getElementById('monthly-view');
  if (Object.keys(byMonth).length === 0) { el.innerHTML = '<div class="empty">Ingen udgifter registreret endnu</div>'; return; }

  el.innerHTML = Object.keys(byMonth).sort((a,b) => b-a).map(m => {
    const data = byMonth[m];
    const catRows = CATS.filter(c => data.cats[c]).map(c => `
      <div class="month-row">
        <span><span class="cat-dot" style="display:inline-block;width:8px;height:8px;border-radius:50%;background:${CAT_COLORS[c]};margin-right:6px"></span>${CAT_ICONS[c]} ${c}</span>
        <span>${fmt(data.cats[c])}</span>
      </div>`).join('');
    return `<div class="month-block card">
      <div class="month-heading">
        <span>${MONTHS[m]}</span>
        <span class="month-total">${fmt(data.total)}</span>
      </div>
      ${catRows}
    </div>`;
  }).join('');
}

// ── MODAL ──
function openAddModal() {
  document.getElementById('modalTitle').textContent = 'Tilføj udgift';
  document.getElementById('editId').value = '';
  document.getElementById('expDate').value = new Date().toISOString().split('T')[0];
  document.getElementById('expCat').value = 'Bil';
  document.getElementById('expAmount').value = '';
  document.getElementById('expDesc').value = '';
  document.getElementById('expenseModal').classList.add('open');
}
function openEditModal(id) {
  const e = getExpenses().find(x => x.id === id);
  if (!e) return;
  document.getElementById('modalTitle').textContent = 'Rediger udgift';
  document.getElementById('editId').value = id;
  document.getElementById('expDate').value = e.date;
  document.getElementById('expCat').value = e.cat;
  document.getElementById('expAmount').value = e.amount;
  document.getElementById('expDesc').value = e.desc || '';
  document.getElementById('expenseModal').classList.add('open');
}
function closeModal() { document.getElementById('expenseModal').classList.remove('open'); }

function saveExpense() {
  const id = document.getElementById('editId').value;
  const date = document.getElementById('expDate').value;
  const cat = document.getElementById('expCat').value;
  const amount = parseFloat(document.getElementById('expAmount').value);
  const desc = document.getElementById('expDesc').value.trim();
  if (!date || !cat || isNaN(amount) || amount <= 0) { alert('Udfyld dato, kategori og beløb.'); return; }

  let expenses = getExpenses();
  if (id) {
    expenses = expenses.map(e => e.id === id ? { ...e, date, cat, amount, desc } : e);
  } else {
    expenses.push({ id: Date.now().toString(36) + Math.random().toString(36).slice(2,6), date, cat, amount, desc });
  }
  saveExpenses(expenses);
  closeModal();
  renderAll();
}

function deleteExpense(id) {
  if (!confirm('Slet denne udgift?')) return;
  saveExpenses(getExpenses().filter(e => e.id !== id));
  renderAll();
}

// ── EXPORT CSV ──
function exportCSV() {
  const expenses = getExpenses();
  if (!expenses.length) { alert('Ingen udgifter at eksportere.'); return; }
  const rows = [['Dato','Kategori','Beløb','Beskrivelse']];
  expenses.sort((a,b) => a.date.localeCompare(b.date)).forEach(e => {
    rows.push([e.date, e.cat, e.amount.toString().replace('.',','), e.desc || '']);
  });
  const csv = rows.map(r => r.map(v => '"'+v.replace(/"/g,'""')+'"').join(';')).join('\r\n');
  const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = 'udgifter.csv'; a.click();
  URL.revokeObjectURL(url);
}

function confirmClear() {
  if (confirm('Er du sikker på, at du vil slette ALLE data? Dette kan ikke fortrydes.')) {
    localStorage.clear();
    renderAll();
    loadSetup();
    showFeedback('Alle data slettet');
  }
}

// ── HELPERS ──
function fmt(n) {
  return new Intl.NumberFormat('da-DK', { style: 'currency', currency: 'DKK', maximumFractionDigits: 0 }).format(n);
}
function fmtDate(d) {
  const [y,m,day] = d.split('-');
  return `${day}. ${MONTHS[parseInt(m)-1].slice(0,3).toLowerCase()}. ${y}`;
}
function renderAll() {
  const cur = document.querySelector('.page.active')?.id?.replace('page-','');
  if (cur === 'overblik') renderOverblik();
  if (cur === 'udgifter') renderExpenses();
  if (cur === 'maaned') renderMonthly();
  document.getElementById('topYear').textContent = getSetup().year;
}
function showFeedback(msg) {
  const d = document.createElement('div');
  d.textContent = msg;
  Object.assign(d.style, {
    position:'fixed', top:'70px', left:'50%', transform:'translateX(-50%)',
    background:'#2ecc8a', color:'#fff', padding:'10px 20px', borderRadius:'20px',
    zIndex:'999', fontSize:'0.85rem', fontFamily:'monospace', boxShadow:'0 4px 16px rgba(0,0,0,0.4)',
    transition:'opacity .5s'
  });
  document.body.appendChild(d);
  setTimeout(() => { d.style.opacity='0'; setTimeout(() => d.remove(), 500); }, 1800);
}
